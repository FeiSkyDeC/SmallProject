richTextBox富文本框属性和方法

using System.IO; //导入读写文件命名空间

⑴ 富文本框读文本
richTextBox1.LoadFile(string path, RichTextBoxStreamType.PlainText); //读txt格式文本
richTextBox1.LoadFile(string path, RichTextBoxStreamType.PlainText); //读rtf、doc格式文本

⑵ 富文本框写文本
richTextBox1.SaveFile(saveFileDialog1.FileName, RichTextBoxStreamType.PlainText); // 保存为txt格式文件
richTextBox1.SaveFile(saveFileDialog1.FileName,RichTextBoxStreamType.RichText); // 保存为rtf、doc格式文件 /p>

(3) 清除富文本框文本
richTextBox1.Clear();

(4) 富文本文本操作
选中的文本：richTextBox1.SelectedText;
删除选中的文本：richTextBox1.SelectedText="";
复制选中的文本：string tmp=richTextBox1.SelectedText; /p>

(5) 打开外部文件
using System.Diagnostics; //导入命名空间，打开外部文件操作
Process P = new Process();
P.StartInfo.FileName = Application.StartupPath + "\\help.htm";
P.StartInfo.Verb = "Open";
P.Start();
